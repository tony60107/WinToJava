choose case dwo.name
	case 'ls_mwb_no2'		
 		 messagebox('錯誤~','主號檢查碼邏輯錯誤~,請檢查後再收費')
		 setitem(row,'ls_mwb_no2','')
		 return 1
	case 'ls_comp_id'
		if f_chk_idno(data) <> 0 then 
			messagebox('錯誤~', '統一編號檢查碼輸入錯誤~')
			setitem(row,'ls_comp_id','')
			return 1
		end if	
		select full_name    , address    , nvl(delete_mark,'N')
		  into :ls_full_name, :ls_address, :ls_delete_mark
		  from customer
		 where inc_id = :data;
		if isnull(ls_full_name) then ls_full_name = ''
		if len(trim(ls_full_name)) = 0 then 
			messagebox('訊息','查無此客戶資料')
			setitem(row,'ls_comp_id','')
			return 1
		end if
		if isnull(ls_delete_mark) then ls_delete_mark = ''
		if trim(ls_delete_mark) = 'Y' then
			messagebox('錯誤~', '此客戶統一編號已被註銷')
			setitem(row,'ls_comp_id','')
			return 1
		end if
	case "ls_title"
		if isnull(data) then data = ''
		if pos(data,'\') > 0 or pos(data,'"') > 0 then
		   messagebox("訊息",'開立發票公司全銜有特殊字元[\或是"]請刪除')
			return 1
		end if
		
	case 'ls_return_type'		
		messagebox('錯誤~', '優退方式只能為線上或後退')
		return 1
	case 'ls_prefer_flag'
		messagebox('錯誤~', '優惠方案只能修改為個案')
		return 1		
	case 'ls_other_type'
		return 1
	case 'ls_trans_fee'
		messagebox('錯誤~', '轉運費率只能輸入 Y 或 N')
		return 1
	case 'ls_gco_date'
		if isnull(data) then data = ''
		if f_chk_fdate(data) <> 0 or len(trim(data)) = 0 then
			messagebox('錯誤~', '輸入之出倉日期無效')
		else
			messagebox('錯誤~', '出倉日期不可小於進倉日期')
		end if
		return 1
	case 'ls_account_no'
		ls_mwb_no  = getitemstring(1,'ls_mwb_no')
		ls_mwb_no2 = getitemstring(1,'ls_mwb_no2')
		ls_hwb_no  = getitemstring(1,'ls_hwb_no')
		if isnull(ls_hwb_no)        then ls_hwb_no = ''
		if len(trim(ls_hwb_no)) = 0 then ls_hwb_no = ' '
		if getitemstring(row,'ls_charge_type') = '1' and f_force_pay_cash(gs_uinfo.u_cargo_loca,ls_mwb_no,ls_mwb_no2,ls_hwb_no,'T') = 'YES' then
			messagebox("訊息","此票貨物已申請現金方式支付倉租，"+char(13)+char(10)+"不可輸入記帳客編【"+data+"】")
		else
			select inc_id      , 
					 cust_attribute
			  into :ls_inc_id    ,
					 :ls_cust_attribute
			  from registered_customer
			 where cargo_location =:gs_uinfo.u_cargo_loca
				and register_no = :data
				and effect_date <= to_char(sysdate,'yyyymmdd')
				and cancel_date >= to_char(sysdate,'yyyymmdd') ;
			if isnull(ls_inc_id) then ls_inc_id = ''
			if isnull(ls_cust_attribute) then ls_cust_attribute = ''
			if ls_cust_attribute = 'A' then 
				if getitemstring(row,'ls_charge_type') = '1' then messagebox('錯誤~','航空公司轉口倉租不可使用線上收費!!')
			end if
			if len(trim(ls_inc_id)) = 0 then messagebox('訊息','記帳客戶錯誤~或已過期!!')
		end if
			
		setitem(row,'ls_account_no','')
		setitem(row,'ls_pay_type','C')
		setitem(row,'ls_comp_id','')
		return 1
	case 'ls_box_id'		
	   messagebox('錯誤~','承攬業~或航空公司代號錯誤~!!')
		return 1
