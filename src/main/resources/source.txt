update ar_trade 
       set close_flag     = 'Y'
     where cargo_location = :gs_uinfo.u_cargo_loca
       and tran_date      = :ls_tran_date
       and seq            = :ls_seq
       and nvl(close_flag,'N') = 'N'
          ;  
    if sqlca.sqlcode <> 0 then 
        rollback ;
        
        json.result_message 加入 messagebox('訊息','更新應收帳款收回資料之過帳狀態失敗 '+'SQLCODE:'+string(sqlca.sqlcode)+' '+SQLCA.SQLErrText)
        return json
    end if
